use config::{Config, Environment, File};
use std::env;
use serde_derive::Deserialize;

#[derive(Debug, Deserialize)]
#[allow(unused)]
struct Database {
    url: String,
}

#[derive(Debug, Deserialize)]
#[allow(unused)]
struct Secret {
    key: String,
    token: String,
    version: u8,
}

#[derive(Debug, Deserialize)]
#[allow(unused)]
struct Settings {
    debug: bool,
    database: Database,
    secret: Secret,
}
